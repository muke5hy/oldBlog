Find and delete duplicate rows from Mongodb

Getting all the data db.duplicate.drop(); db.scrip_data.aggregate([ { $group: { _id: { time: "$time", code: "$code", ltp:"$ltp" }, uniqueIds: { $addToSet: "$_id" }, count: { $sum: 1 } }},

{ $match: { count: { $gt: 1 } }}, {"$out": "duplicate"} ],{allowDiskUse:true});

db.duplicate.find({}).sort({count:-1}).forEach( function(data){ if (data.uniqueIds.length > 1){ data.uniqueIds.shift(); print(data.uniqueIds.length); db.scrip_data.remove({"_id":{$in:data.uniqueIds}}); } })
